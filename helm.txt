
helm repo list
helm repo add <REPOSITORY_NAME> <REPOSITORY_URL>
helm repo add bitnami https://charts.bitnami.com/bitnami

helm search repo bitnami

# Show all installs
helm list -a

helm create <chart>

helm install <release> <chart>
helm install <release> --debug --dry-run <chart>
helm uninstall <release>

helm get manifest <release>

helm diff revision <chart> <rev1> <rev2>
helm upgrade <release> <chart>
helm rollback <release> <revision>

# Checks
helm lint <chart>
helm template <chart>
helm template <release> <chart> -f path/to/values.yaml
- or
helm template --replace <chart> --values path/to/values.yaml


helm dependency update <dir>
helm package <dir>
helm push-artifactory <chart> <repo>
