kubectl create ns ingress-nginx
helm install ingress-nginx  ingress-nginx/ingress-nginx --namespace ingress-nginx

---
# Test an ingress, get the ingress address
kubectl get ingress -n <namespace>

# Use curl with the ingress address and the host end-point
curl -v http://<ingress-address> -H "Host: <end-point>"
ie:
curl -v http://172.16.33.131 -H "Host: hello.local"
